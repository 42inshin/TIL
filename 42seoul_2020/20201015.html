# 1. substr

char	*ft_substr(char const *s, unsigned int start, size_t len)
{
	char	*ptr;
	size_t	i;

	ptr = (char *)malloc(sizeof(char) * len);
	if (!ptr)
		return (NULL);
	i = 0;
	while (i < len && s[start])
	{
		ptr[i] = s[start];
		start++;
		i++;
	}
	return (ptr);
}
  • substr 함수는 문자열 s 에서 start index 부터 최대 크기 len 만큼을 할당된 메모리에 복사하는 함수이다.
  • 메모리 할당에 실패하면 NULL 반환.

# unit-test 에서 오류처리

// unit test 오류 처리. 아래 코드 추가함.
if (!s)
		return (NULL);
if (start >= ft_strlen(s))
		return (ft_strdup(""));
  • 문자열 s가 주소값이 없을 경우, NULL 포인터를 반환해야 한다.
  • start 가 문자열 s의 길이보다 클 경우, ft_strdup("") 로 빈문자열을 반환해야 오류가 안남.
    • return (NULL); 과 return(ft_strdup("")); 의 차이
      • return (NULL); : 주소값이 NULL, 즉 주소값이 없으므로 s[0] 출력 시 세그먼트 폴트.
      • return(ft_strdup("")); : s[0] = '\0' 과 동일. 문자열 끝이 바로 검색됨.
    • char *s = NULL : s[0] 존재 안함. 세그먼트 폴트.
    • char *s = '' : s[0] = '\0'

# 2. strjoin

char	*ft_strjoin(char const *s1, char const *s2)
{
	char	*ptr;
	size_t	total_len;
	int		i;
	int		j;

	total_len = ft_strlen(s1) + ft_strlen(s2);
	ptr = (char *)malloc(sizeof(char) * total_len);
	if (!ptr)
		return (NULL);
	i = -1;
	while (s1[++i])
		ptr[i] = s1[i];
	j = -1;
	while (s2[++j])
		ptr[i++] = s2[j];
	return (ptr);
}
  • 문자열 s1, s2의 길이만큼 메모리를 할당하여 s1 뒤에 s2 를 이어붙임.
  • 메모리 할당 실패시 NULL 반환함.

# unit test 오류 수정

// 아래 코드 상단에 추가함.
if (!s1 && !s2)
		return (NULL);
	if (!s1)
		return (ft_strdup(s2));

  • s1, s2 주소값이 null 인 경우, NULL 포인터 반환
  • s1, s2 중 하나만 null 인 경우, 나머지를 메모리 할당하고 반환함.(ft_strdup(나머지 s))

# git pull 에러 해결방법

  • local pc 환경에서 수정을 거친 파일을 지닌 채로는 git pull 을 하려고 해도 오류가 뜬다.
  • 이 pc에서 수정한 것이 최신 수정이라면 push 하고 git pull 하여 나머지를 받는 것도 해결방법.
  • 하지만 현재 올라가있는 파일들이 최신 파일이라면, 지금 local에서 수정된 것 올리기 애매해짐.
  • 해결방법으론 작업중이던 파일을 임시로 저장해두고 현재 브랜치의 상태를 마지막 커밋의 상태로 초기화 하는 것임.
  • 스택에 임시저장하고 초기화 해주는 명령어 : git stash
  • 위 명령어를 쓰고 git pull 을 하면 됨.
  • 임시저장된 스택을 제거하는 명령어 : git stash drop

git stash 참고 : https://gmlwjd9405.github.io/2018/05/18/git-stash.html

# 7. 학습 내용에 대한 개인적인 총평

  • libftest 에서 모두 통과 되더라도 libft-unit-test 에서 걸릴 수 있다.
  • 생각보다 함수를 만드는 것에 예외처리를 해야 될 부분이 많아서 시간을 오래 써야될 것 같다.
  • ft_putchar 의 경우, unicode 에 대한 출력도 테스트기에 나오는데 unicode 출력 방식에 대해서 연구해봐야 될 것 같다.
  • 여러가지 코드를 조금 조금씩 고쳐 나가다 보니, 코드가 정확히 어떤 식으로 구현되어 있는지 모두 기억하기 어렵다.
  • malloc 으로 할당할 때, '\0' 이 들어가는 공간까지 할당해야 되는지 찾아봐야 할 것 같다.
    • 한국에 malloc 을 사용하는 예제를 보면 \0 에 대해서 신경쓰지 않으나..
    • 정확한 정보가 필요하다.
