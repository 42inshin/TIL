# 6. 상세 학습 내용

실제 코딩 시간 : 총 2시간.

# ft_strmapi

char	*ft_strmapi(char const *s, char (*f)(unsigned int, char))
{
	char	*tmp;
	int		i;

	if (!s || !f)
		return (NULL);
	tmp = ft_strdup(s);
	i = 0;
	while (tmp[i])
	{
		tmp[i] = f(i, s[i]);
		i++;
	}
	tmp[i] = '\0';
	return (tmp);
}

# 구현 방법

  • 문자열 *s 의 **"각 문자"**마다 f함수를 실행 할 수 있도록 함수를 구현.
  • 각 문자마다 f함수가 영향을 미칠 수 있도록 구현한다.
  • f 함수의 매개변수는 unsigned int i, char c.
  • *s 와 (*f) 의 관계를 생각해서 이어주는 것이 핵심이다.
  • 이 과정에서 *s 를 담을 *tmp 를 만들고 ft_strdup() 함수를 이용하였음.

# 함수 포인터

  • void (*fp)(); : 반환값자료형 (*함수포인터이름)();
  • 예시
#include <stdio.h>

void hello()     // 반환값과 매개변수가 없음
{
    printf("Hello, world!\n");
}

void bonjour()    // 반환값과 매개변수가 없음
{
    printf("bonjour le monde!\n");
}

int main()
{
    void (*fp)();   // 반환값과 매개변수가 없는 함수 포인터 fp 선언

    fp = hello;     // hello 함수의 메모리 주소를 함수 포인터 fp에 저장
    fp();           // Hello, world!: 함수 포인터로 hello 함수 호출

    fp = bonjour;   // bonjour 함수의 메모리 주소를 함수 포인터 fp에 저장
    fp();           // bonjour le monde!: 함수 포인터로 bonjour 함수 호출

    return 0;
}

// 결과
Hello, world!
bonjour le monde!

# 7. 학습 내용에 대한 개인적인 총평

  • 함수포인터라는 개념에 대해 새롭게 학습하였다. 여전히 배워야 할, 넘어야 할 산들이 많다. 아직 기초에 불과한 지금의 과정을 탄탄히 잘 만들어서 앞으로 나아갈때 어려움 없이 갈 수 있도록 학습에 최선을 다하도록 해야겠다.
  • 내용에 대한 이해와 실습은 엄연히 다른 과정이다. 계속 풀어가면서 사용법에 익숙해지는 시간을 많이 만들자.
  • 앞으로의 과정을 위해서 하는 libft 이지만, 쉽지만 않은 시간들이었다. 미래의 나를 위해 지금의 시간들을 허비하지 않도록 마음을 다잡고 다시 잘해보는 시간을 갖도록 노력하자.
  • 또한, 42서울에 적응하고 사람들과 잘 지낼 수 있는 시간들도, 추억들도 많이 만들도록 마인드를 잘 잡자.
