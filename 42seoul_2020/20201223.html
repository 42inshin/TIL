# 6. 상세 학습 내용

실제 코딩 시간 : 총 3시간.

# 상세 코드

#include "get_next_line.h"

int	get_next_line(int fd, char **line)
{
	static char	*backup;
	char		*tmp;
	char		*nl_ptr;
	char		buf[BUFFER_SIZE + 1];
	int			read_size;

	if (fd < 0 || line == NULL || BUFFER_SIZE <= 0)
		return (-1);
	nl_ptr = NULL;
	while ((read_size = read(fd, buf, BUFFER_SIZE)) > 0)
	{
		buf[read_size] = '\0';
		tmp = ft_strjoin(backup, buf);
		if (backup != NULL)
			free(backup);
		backup = tmp;
		if ((nl_ptr = ft_strchar(backup, '\n')) != 0)
			break ;
	}
	if (read_size < 0)
		return (-1);
	if ((nl_ptr = ft_strchar(backup, '\n')) != NULL)
	{
		*line = ft_strndup(backup, nl_ptr - backup);
		tmp = ft_strdup(nl_ptr + 1);
		free(backup);
		backup = tmp;
		return (1);
	}
	if (backup != NULL)
	{
		*line = backup;
		backup = NULL;
	}
	else
		*line = ft_strdup("");
	return (0);
}

# 다시 짜면서 실수하는 부분

  1. BUFFER_SIZE <= 0 예외처리 빠뜨림.
  2. if ((nl_ptr = ft_strchar(backup, '\n')) != NULL)
    • 위의 read while 문에 값이 안들어가는 경우에도 nl_ptr 이 작동 되어야 하므로 아래 if 문에서 처리해줘야 하는데 위에 있다고 생각하고 안하는 경우가 종종 있음.
    • 코드를 수정하기 위해선 윗 while 문에 이부분까지 같이 넣으면 처리 가능할것임.
  3. *line = ft_strdup(""); 마지막 처리
    • 마지막 부분에 nl_ptr == NULL, backup == NULL 인 경우, 처리도 해줘야함.

# 7. 학습 내용에 대한 개인적인 총평

  • gnl 코드와 필요한 utils 까지 순전히 다 짜다보니 예상보다 시간소요가 많이 걸렸다. 다음 서클에서 시험을 치기 때문에 미리 시험친다 생각하고 연습했는데, 더 준비를 해야 될 것 같다.
  • 평가도중 gnl 의 이중포인터 (**line) 부분에 대해서 설명을 해보려했는데 한마디로 쉽게 설명하기가 까다로웠다. 이중포인터다보니 이중배열로 생각할 수도 있는데, 여기에선 이중배열 때문에 사용하는 것은 아니여서 이부분에 대해서 인지 못하는 분에게 설명하는 것은 무척이나 어려웠다. 남들에게 잘 자세히 설명할 수 있을 정도로 이해도를 높이도록 해야겠다.
    • 변수 값을 바꿔줄 때는 주소값을 이용하여 바꾸는 것처럼, 포인터가 지시하는 값을 바꾸려면 이중포인터에서 접근해서 값을 바꿔줘야 한다. malloc 을 사용해서 계속 새로운 주소값으로 새로운 값이 있는 곳이 바뀌게 되는데, 이 주소값이 변경이 되므로 주소값 변경을 위해, 즉 포인터값 변경을 위해선 접근을 이중포인터에서 해야 된다? 라고 하면 될까나,, 설명을 정확하게 하기 위해서 다시 이부분에 대해 정리를 해야겠다.
