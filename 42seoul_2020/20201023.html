# 6. 상세 학습 내용

실제 코딩 시간 : 총 2시간.

# ft_itoa

#include "libft.h"

int		ft_numlen(int n)
{
	int		i;

	i = 0;
	if (n <= 0)
		i++;
	while (n)
	{
		n /= 10;
		i++;
	}
	return (i);
}

char	*ft_itoa(int n)
{
	char	*str;
	int		num_len;
	int		sign;

	sign = 1;
	num_len = ft_numlen(n);
	if (n < 0)
	{
		n *= -1;
		sign = -1;
	}
	if (!(str = (char *)malloc(sizeof(char) * (num_len + 1))))
		return (NULL);
	str[num_len] = '\0';
	while (--num_len > 0 && n != 0)
	{
		str[num_len] = n % 10 + '0';
		n /= 10;
	}
	if (sign < 0)
		str[0] = '-';
	else
		str[0] = n % 10 + '0';
	return (str);
}
  • KeyPoint :
    • 음수, 양수에 따라 len 을 설정, malloc으로 메모리 할당 후 뒤 에서부터 채워준다.
    • 정수의 길이를 잴 때는 n /= 10을 이용, 나눠지는 갯수로 len을 설정한다.
    • 음수는 - 를 표현해야 하므로 len + 1 해준다.
    • 0 에 대한 len 처리도 해야함.
    • 각 자릿수는 나머지 연산(n % 10 + '0')으로 구하고 뒤 에서부터 채운다.
    • 음수를 표현하기 위해선 양수(절대값)로 바꿔준 후 나머지 연산을 하고, str[0] 에 '-' 를 넣어준다.

# 7. 학습 내용에 대한 개인적인 총평

  • ASCII 코드로 표현하기 위해서는 양의 정수로 바꿔줘야 하므로, 그때의 예외 처리를 하는데 생각보다 많은 시간을 소요했다.
  • 최근 개인적인 일처리 등으로 시간을 쪼개서 학습을 하는 중인데, 지속적으로 계속 하는 것이 중요하다 생각이 들며, 비록 적은 시간이지만 꾸준히 하고 있는 나에게 잘하고 있다고 말해주고 싶다.
  • malloc 을 사용하면, free 를 해줘야 한다고 알고 있는데, 이러한 함수를 실행했을 때는 메모리 처리를 어떤식으로 해야하는지에 대해 알아볼 필요가 있다.
  • 예전에 비해 확실히 코드 구조를 짜는데 이해력은 높아졌으나, 점차 더 깔끔하고 빠르게 동작할 수 있는 코드를 구성하는 목표로 학습을 해나가자.
