# gnl 문제 및 해결방법

# 문제점

  • gnl 에서 BUFFER_SIZE가 1~6 까진 오류없이 작동되었으나, 7부터 작동이 안되는 문제.
  • 이 문제를 고치기 위해서 계속 코드를 수정하며, 다른 카뎃 분들의 코드와 비교를 하면서 문제를 수정하려고 했으나 어디서 문제점이 생기는지 찾기가 어려웠었다.
static int	get_return_value(char **save, char **line, char *l_ptr)
{
	char *tmp;

	if ((l_ptr != NULL) // 이 부분이 문제였음!
	{
		*line = ft_strndup(*save, l_ptr - *save);
		tmp = ft_strdup(l_ptr + 1);
		free(*save);
		*save = tmp;
		return (1);
	}
	if (*save != NULL)
	{
		*line = *save;
		*save = NULL;
		return (0);
	}
	*line = ft_strdup("");
	return (0);
}

int			get_next_line(int fd, char **line)
{
	static char	*save[FD_MAX];
	char		buf[BUFFER_SIZE + 1];
	int			read_size;
	char		*l_ptr;
	char		*tmp;

	if (fd < 0 || line == NULL || BUFFER_SIZE <= 0)
		return (-1);
	while ((read_size = read(fd, buf, BUFFER_SIZE)) > 0)
	{
		buf[read_size] = '\0';
		tmp = ft_strjoin(save[fd], buf);
		if (save[fd] != NULL)
			free(save[fd]);
		save[fd] = tmp;
		if ((l_ptr = ft_strchr(save[fd], '\n')) != 0)
			break ;
	}
	if (read_size < 0)
		return (-1);
	return (get_return_value(&save[fd], line, l_ptr));
}
  • 최상단의 코드에서 l_ptr != NULL 일 경우에 해당하는 구문이 문제였다.
  • 문제가 되는 이유는 파일을 다 읽어서 read_size 가 0이 될 경우, save[fd] 에는 '\n' 이 있는데도 불구하고 l_ptr 값을 새로 부여 받지 않아 발생했다. \n 이 있지만 해당 if문에 엉뚱한 l_ptr 주소값이 들어가면서 무한루프에 빠지게 된다. 그러면서 세그먼트 에러가 나는 것이다.

# 7. 학습 내용에 대한 개인적인 총평

  • 혼자서 오류를 잡기위해서 전날 하루종일 보았지만, 다시 코드를 작성하고 전반적인 구현방법을 바꾸면서 진행하였으나 새로운 오류들이 생기곤했다. 하지만 슬랙을 통해서 다른분들과 코드를 공유하여 문제점을 찾아보니 간단하게 끝낼 수 있던 문제였다. 역시, 42서울에선 동료학습이 꼭 필요하다고 생각이 드는 시간이었다. 혼자서 어떻게든 해보자라는 생각으로 붙잡고 있던 시간들도 뜻깊은 시간이지만, 함께 하는 이 상황을 최대한 적극적으로 사용해보도록 해야겠다.
  • 지금 코드와 그전에 짠 코드들도 그 방법에 따라서 구현할 수 있도록 구상해보는 작업을 해보도록 해야겠다.
