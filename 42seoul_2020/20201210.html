# 코드 구현

int	get_next_line(int fd, char **line)
{
	static char	*save[FD_MAX];
	char		buf[BUFFER_SIZE + 1];
	int			read_size;
	char		*newline_ptr;
	char		*tmp;

	if (fd < 0 || line == NULL || BUFFER_SIZE <= 0)
		return (-1);
	newline_ptr = NULL;
	while ((read_size = read(fd, buf, BUFFER_SIZE)) > 0)
	{
		buf[read_size] = '\0';
		if ((newline_ptr = ft_strchr(buf, '\n')) == 0)
			save[fd] = ft_strjoin(save[fd], buf);
		else
			break ;
	}
	if (read_size < 0)
		return (-1);
	if ((newline_ptr = ft_strchr(buf, '\n')) != NULL)
	{
		tmp = ft_strndup(buf, newline_ptr - buf);
		*line = ft_strjoin(save[fd], tmp);
		save[fd] = ft_strdup(newline_ptr + 1);
		return (1);
	}
	else if (save[fd])
	{
		*line = save[fd];
		save[fd] = NULL;
		return (0);
	}
	*line = ft_strdup("");
	return (0);
}
  • 어제의 코드로 가장 문제가 되던 코드를 해결 했지만, GNL-lover 테스터에서만 제대로 동작하고 다른 테스터에선 오류가 나는 상황이어서 새롭게 코드를 짰다.
  • main 함수를 만들어서 gnl 이 제대로 동작하는지 확인하고 printf를 통해서 각각의 코드 사이사이마다 값이 제대로 들어가고 있는지 확인을 하였다.
  • 중간에 코드가 꼬이는 부분이 있어서 계속해서 오류를 찾기 위해서 머리속에서 코드를 동작시켜 보았다.

# 7. 학습 내용에 대한 개인적인 총평

  • 결론적으론, 위 코드는 buf에서 \n을 찾다보니 save에 \n이 있는 버퍼 구간이 들어가지 않으면서 오류가 나는 것이 였는데, 한번 안보이기 시작하면 계속 그 코드를 찾아내지 못하는 상황이 있었다.
  • 이 부분을 고치자 다른 오류가 생성되었다. 내일 고치기로 결정함.
  • 디버그 프로그램을 이용하는 방법에 대해서 찾아봐야 될 것 같다. 머리속에서 코드 동작을 굴리다보니 조금만 멍때리면 앞에서 생각한부분을 잊곤한다. 시간이 너무 소요가 되므로 디버깅에 대해 검색해봐야겠다.
