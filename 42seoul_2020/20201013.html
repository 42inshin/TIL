# 1. memmove

void	*memmove(void *dst, const void *src, size_t len)
{
	unsigned char *cp_dst;
	unsigned char *cp_src;

	if (dst < src)
	{
		cp_dst = (unsigned char *)dst;
		cp_src = (unsigned char *)src;
		while (len--)
			*cp_dst++ = *cp_src++;
	}
	else
	{
		cp_dst = (unsigned char *)(dst + len - 1);
		cp_src = (unsigned char *)(src + len - 1);
		while (len--)
			*cp_dst-- = *cp_src--;
	}
	return (dst);
}
  • man 해설 : memmove() 함수는 문자열 src 에서 문자열 dst 에 len 바이트를 복사합니다. 2 개의 캐릭터 라인이 오버랩 될 수 있지만, 사본은 항상 파괴되지 않는 방식으로 이루어집니다.
  • 반환 값 : 함수는 dst 의 원래 값을 반환합니다.
  • 데이터가 겹칠 경우, "마치" 복사할 문자열을 버퍼에 할당한 것처럼 구현됨.
  • "마치" 의 실제 구현 방법은 dst , src 의 주소 값 순서에 따라 복사 방향을 다르게 해주는 것.
  • dst 가 src 보다 앞 에 있는 경우, 순방향으로 복사를 해도 아무런 영향이 없지만, dst 가 src 보다 뒤 에 있는 경우, 순방향으로 복사하면 원본이 아닌 복사했던 부분을 다시 복사하게 될 수 있다.
  • 그렇기 때문에, dst > src 일때 역방향으로 복사가 되도록 코드를 구현해주면 해결된다.

# 2. memcpy , memmove 비교

  • memcpy는 무조건 순방향으로만 복사함.

  • memmove는 겹쳐진 메모리에 대해서 원본이 파괴되지 않도록 복사 방향을 지능적으로 선택함. 마치, "임시공간" 에 저장하여 복사하는 것처럼 구현되어 있음.

    #include <stdio.h>
#include <string.h>
#include "libft.h"

int main() {
	char str[] = "memmove can be very useful......";
	char str1[] = "memmove can be very useful......";
	char str2[] = "memmove can be very useful......";

	ft_memcpy(str + 20, str + 15, 11);
	puts(str);
	memcpy(str1 + 20, str1 + 15, 11);
	puts(str1);
	memmove(str2 + 20, str2+ 15, 11);
	puts(str2);
	return 0;
}

    # 결과

    // gcc 컴파일
>> ./a.out
memmove can be very very very v. // ft_memcpy
memmove can be very very useful.  // memcpy
memmove can be very very useful.  //memmove
    • gcc 로 컴파일 할 시, memcpy 와 memmove 는 차이가 없었다.
    • 직접 만든 ft_memcpy 와 확연한 차이를 보인다.
    // clang 컴파일
>> ./a.out
memmove can be very very very v.  // ft_memcpy
memmove can be very very usey u.  // memcpy
memmove can be very very useful.  // memmove
    • clang 컴파일러에선 결과 값이 달라졌다.
    • 컴파일러 종류에 따라 결과 값이 달라질 수 있다는 것을 알게 됨.
    • 하지만, 예상했던 결과(ft_memcpy)와 다른 결과가 나옴.

  • 컴파일러에 따라 memcpy 함수를 최적화 할 수 있으므로, memcpy 사용에 유의할 필요가 있음.

  • 의문점 : len 값을 int i 로 빼서 값을 주면 clang 에서도 정상작동 한다. 이 부분은 아직 이해가 되지 않는 상황임....


# 7. 학습 내용에 대한 개인적인 총평

  • memmove 와 memcpy 에 대해 자세히 알아보면서, 컴파일러에 따라서 결과가 달라지는 것을 확인할 수 있었다.
  • 또한, 함수에 직접 숫자 (11) 를 넣었을 때와 int i = 11; 의 결과가 달라지는 것도 확인할 수 있었다.
  • 이 문제의 경우, 각 컴파일러의 작동 방법에 대해서 알아봐야 할 부분인 듯하다.
  • 한번의 밤샘이 일주일의 스케줄을 망친다.
