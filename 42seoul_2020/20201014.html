# 1. memchr

void	*ft_memchr(const void *s, int c, size_t n)
{
	unsigned char	value;
	unsigned char	*str;

	value = c;
	str = (unsigned char *)s;
	while (n-- && *str)
	{
		if (*str == value)
			return (str);
		str++;
	}
	return (NULL);
}
  • memchr : byte 문자열에서 바이트 찾기.
  • 문자열 *s 에서 n 길이 만큼 c(converted to an unsigned char) 문자를 찾는 함수.
  • int c 로 받지만 unsigned char 변수로 받으면 형변환 따로 하지 않아도 괜찮음.
  • return :
    • 문자열 시작부터 n bytes 안에 문자 c 가 존재하면 그 위치의 포인터를 반환함.
    • n bytes 안에 c 가 없다면 NULL 을 반환한다.

# 2. memcmp

int	ft_memcmp(const void *s1, const void *s2, size_t n)
{
	unsigned char	*ptr1;
	unsigned char	*ptr2;

	ptr1 = (unsigned char *)s1;
	ptr2 = (unsigned char *)s2;
	while (*ptr1 && *ptr2 && n)
	{
		if (*ptr1 != *ptr2)
			return (*ptr1 - *ptr2);
		ptr1++;
		ptr2++;
		n--;
	}
	if (n == 0)
		return (0);
	if (!*ptr1)
		return (-1);
	return (1);
}
  • ft_memcmp : byte 마다 s1, s2 를 비교하는 함수.
  • 두 문자열 다 n byte 보다 길어야 함.
  • s1 과 s2 값을 n byte 만큼 비교하여 같으면 0 반환, 다르면 *s1 - *s2 을 반환.
  • s1 이 클 경우 양수, s2가 클 경우 음수
  • 길이가 0 인 문자열은 항상 동일합니다. (Zero-length strings are always identical.)
    • libft test 를 돌렸을 때, 문자열이 NULL 일때 문제가 발생되어, if (!*ptr1) 의 리턴값을 -1, 1 로 수정하였음.

# 3. toupper

int		ft_toupper(int c)
{
	if (c >= 'a' && c <= 'z')
		return (c + 'A' - 'a');
	return (c);
}
  • int c 값이 소문자 일 경우, 대문자 로 반환하라.
  • 그렇지 않으면 int c 값 그대로 반환.

# 4. tolower

int		ft_tolower(int c)
{
	if (c >= 'A' && c <= 'Z')
		return (c + 'a' - 'A');
	return (c);
}
  • int c 값이 대문자 일 경우, 소문자 로 반환하라.
  • 그렇지 않으면 int c 값 그대로 반환.

# 5. calloc

void	*ft_calloc(size_t count, size_t size)
{
	void *ptr;

	if (!(ptr = malloc(size * count)))
		return (NULL);
	ft_bzero(ptr, size * count);
	return (ptr);
}
  • calloc : size byte 크기의 데이터 타입을 count 개 저장할 수 있도록 메모리를 할당.
  • 즉, size * count 바이트 만큼 메모리 할당.
  • 할당된 메모리의 값은 0 으로 채웁니다.
  • calloc 은 heap에 메모리를 할당함.
  • return :
    • 메모리가 성공적으로 할당 되면 주소값을 반환함.
    • 메모리 할당에 오류가 있다면 NULL 포인터를 반환함.

# 6. strdup

char	*ft_strdup(const char *s1)
{
	char	*ptr;
	int		len;
	int		i;

	len = ft_strlen(s1);
	if (!(ptr = (char *)ft_calloc(len + 1, sizeof(*s1))))
		return (NULL);
	i = -1;
	while (s1[++i])
		ptr[i] = s1[i];
	return (ptr);
}
  • strdup : 문자열 s1 복사를 위해 충분한 메모리를 할당하고, 그 주소값을 반환.
  • 그 주소값은 free() 함수의 매개변수로 사용될 수 있다.
  • 메모리 부족으로 할당이 안된 경우, NULL 반환.
