# 코드 구현

int	get_next_line(int fd, char **line)
{
	static char	*save[FD_MAX];
	char		buf[BUFFER_SIZE + 1];
	int			read_size;
	char		*nl_ptr;
	char		*tmp;

	if (fd < 0 || line == NULL || BUFFER_SIZE <= 0)
		return (ERROR);
	nl_ptr = NULL;
	// 버퍼에 텍스트 읽고, read_size 갱신하고 널문자 처리.
	// '\n' 을 만나는 처리해줘야함.
	while ((read_size = read(fd, buf, BUFFER_SIZE)) > 0)
	{
		buf[read_size] = '\0'; // 문자열 끝 처리
		// newline 확인
		if ((nl_ptr = ft_strchr(buf, '\n')) == 0)
			save[fd] = ft_strjoin(save[fd], buf); // '\n'이 없다면 buf를 메모리할당하여 save에 저장.
		else
			break ;
			// newline 만난다면 buf~\n까지 malloc 하고 line과 연결함.
			// newline 만났으니 while 문 종료. nl_ptr만 기억하면 될듯.
	}
	if (read_size < 0)
		return (ERROR);
	// 빈 문서를 읽어왔다면,
	// buf 시작부터 \n 까지 tmp에 넣어서 save[fd]에 합치자.
	tmp = ft_substr(buf, 0, (nl_ptr - buf));
	save[fd] = ft_strjoin(save[fd], tmp);
	if (nl_ptr != NULL) // newline이 있었다면 다음줄로 넘어가자.
	{
		*line = ft_strdup(save[fd]);
		free(save[fd]);
		save[fd] = ft_substr(nl_ptr, 1, 0);
		return (READ_OK);
	}
	if (save[fd]) // newline 없고 종료될 경우,
	{
		*line = save[fd];
		save[fd] = NULL;
	}
	else
		*line = ft_strdup("");
	return (END);
}

# 리턴값

  • 1: 1라인을 읽었다.
  • 0: 끝까지 읽었다.
  • -1: 오류가 났다.

# gnl 동작 순서

  1. open 을 통해 읽은 “file.txt”을 fd 에 담는다.
  2. 만약에 fd 가 지정되지 않는 상황 예외 처리.
  3. read 함수를 통해서 fd를 버퍼 사이즈 만큼 buf 에 읽는다. buf[buf_size]
  4. buf를 static 변수인 save에 담는다.
  5. 만약 버퍼에서 \n 을 만났다면 save에 \n 앞까지 버퍼를 담아 line 에 연결해주고, \n 뒤부터는 다시 static 변수인 save 에 담아준다. 그리고 gnl이 다시 실행될 수 있도록 1을 반환한다.
  6. 만약 \n 을 못 만났다면 3번으로 돌아가서 save에 합쳐준다.
  7. read 에서 EOF에 도달한 경우, save[fd] 안 \n의 여부에 따라 gnl의 리턴 값이 달라진다.

# 개행 문자 처리 과정

  1. \n 을 만났다.
    • \n 전까지를 새로 메모리할당 후 저장해서 line 에 넣어줌
    • \n 뒤에부터의 주소값은 static 변수인 save[fd] 저장.
  2. \n 을 안만났다.
    • 줄이 아직 남아있다. -> save[fd] = strjoin(buff);
    • 줄이 끝났다. buf -> *line = save[fd];

# 문제점

  • 버퍼에 개행문자가 존재하면 save[fd]에 담지 않고, 이후에 처리를 해주었는데 만드는 과정에서 계속 오류가 나는 상황이 발생한다.
  • 메모리 누수에 관한 처리를 제대로 하지 않아 메모리 누수가 발생한다.

# 필요한 함수

strchr : \n 과 만나는거 검사 strjoin : 기존의 문자열 뒤에 이어주기 (malloc) strdup: 아무것도 없는 경우 문자열 생성 (malloc) strlen: 문자열 길이 확인 substr: 개행부분 앞 뒤 처리할때 사용.

# 7. 학습 내용에 대한 개인적인 총평

  • 오류를 수정하면서 코드에 대한 이해도가 더 올라갔다. 처음에 gnl 동작방법에 대해서 생각했던 부분들에서 논리적으로 오류가 나는 부분들이 있었고, 계속해서 주석을 달아가면서 구현을 해보았다.
  • 생각하는 것과 구현하는 부분에 있어서 아직까지 속도가 나지 않아서 답답하기도 하고, 방금까지 생각했던 부분들을 기억하지 못해서 다시금 멍해지는 경우가 종종있었다. 작업을 하면서 계속해서 주석을 달아서 생각하고 있는 부분을 기록하는 것이 중요하다는 사실을 느꼈다.
  • 앞으로 printf 부터 계속 코드 길이가 길어질텐데, 코드 읽는 속도가 늘 수 있도록 계속 익숙해지도록 읽어나가고 기록하자.
  • 경열님과 슬랙콜을 통해서 함께 gnl 코드 오류 수정을 했다.
  • 처음 save를 초기화하는 코드가 들어가있었는데 그때문에 다시 gnl이 실행 될때마다 담겨있던 save의 주소값을 잃어버려 메모리 누수와 값이 제대로 나오지 않는 문제가 있었다. 한참을 이유를 못찾다가 결국 그 코드를 제거하니 당장의 문제가 해결되었다. 간단하지만 논리적 오류를 범했을때 코드가 제대로 동작하지 않는 경우, 오류를 찾기 너무나도 힘들었다. 이러한 상황을 항상 조심하려면 전반적인 구조를 완전히 이해하면서 코드를 짜야하는데, 코드를 짜면서 구조를 다시 잡아가는 경우가 많기 때문에 알고리즘의 실수를 최소화 하기 위해서 계속해서 코드를 복기해봐야 할 것 같다.
