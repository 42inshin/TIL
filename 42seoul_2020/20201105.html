# ft_lstiter

void	ft_lstiter(t_list *lst, void (*f)(void *))
{
	if (lst && f)
	{
		while (lst != NULL)
		{
			f(lst->content);
			lst = lst->next;
		}
	}
}
  • lst 를 반복하면서 함수 f 를 적용 시키는 함수.
  • lstdelone 과 유사하게 함수 f를 lst->content 에 적용 시키고, 다음 노드로 넘어가면서 반복하면 된다.
t_list	*ft_lstmap(t_list *lst, void *(*f)(void *), void (*del)(void *))
{
	t_list *start;
	t_list *new;
	t_list *curr;

	if (l == NULL || f == NULL)
		return (NULL);
	if (!(start = ft_lstnew(f(l->content))))
		return (NULL);
	curr = start;
	l = l->next;
	while (l)
	{
		if (!(new = ft_lstnew(f(l->content))))
		{
			ft_lstclear(&start, d);
			return (NULL);
		}
		curr->next = new;
		curr = new;
		l = l->next;
	}
	return (start);
}
  • lst 에서 부터 함수 f 를 적용하면서 새로운 연결 리스트를 만드는 함수.
  • 중간에 메모리가 부족해 할당이 되지 않는 경우, 만들었던 처음부터 다 free 해준다.
  • l->next 로 이동하면서 f 를 적용한 새로운 리스트를 만들고, start 에 이어준다.
  • 기존에 있는 리스트에서 변경이 아닌, 새롭게 만들어준다는 점을 기억하자.

# 7. 학습 내용에 대한 개인적인 총평

  • 연결리스트에 대해서 계속 학습하면서 간단하지만 이해가 어려운 부분들이 있어서 많은 시간을 소요했다. 그렇지만 계속해서 늘어가고 있음은 분명하니 꾸준하게 계속 책상에 앉아 코딩하는 작업을 늘리도록 해야겠다.
  • 여전히 전공자들과 비교한다면 하늘과 땅의 차이만큼 실력 차가 있다. 더 열심히 더 꾸준히 하자.
  • 역시나 동료와 함께 학습을 하면서 이 과정에 대해 즐거움을 느낄 수 있었다. 혼자서 열심히 하는 것은 한계가 있고, 계속해서 지치는 경향이 많다. 즐거운 마음과 내가 성장하고 도움을 줄 수 있는 사람이 된다는 믿음. 그러한 것들이 시너지로 작용해서 우리가 꿈을 이룰 수 있을 것이다. 앞으로 계속 더 고된 과정을 겪게 되겠지만 사람들로 하여금 그 과정을 즐겁게 헤쳐나가길 희망한다.
