# 1. memset

void *ft_memset(void *b, int c, size_t len)
{
  size_t			i;
  unsigned char	*str;

  i = 0;
  str = b;
  while (i++ < len)
  	*str++ = (unsigned char)c;
  return (b);
}
  • memset 함수는 어떤 메모리의 시작점(b)부터 연속된 범위(len)를 어떤 값(c)으로(바이트 단위) 모두 지정하고 싶을 때 사용하는 함수이다.
  • 초기화 할 값인 c는 인자로는 int로 전달되지만 함수에서는 unsigned char로 형변환 되어서 사용해야 한다.(man memset)
  • 1 byte 단위로 초기화 한다.
    그러므로 char형 배열은 1byte로 괜찮지만, int형 배열은 4byte라서 초기화 할 때 주의해야 한다.

    예를 들면 arr가 int형 일 때, ft_memset(arr, 1, sizeof(arr))을 호출하면 arr 배열이 모두 1 로 초기화 된다고 생각하는데, 실제로는 배열이 모두 16843009 로 초기화 된다. 16843009의 2진수 표현은 0001 00000001 00000001 00000001 이다. 즉, int형 arr 배열은 1바이트(8비트)당 1로 초기화 된 것이다. 바이트가 딱 떨어지게 초기화 되지 않기 때문에 int형 배열의 요소를 1로 초기화 할 수 없다. 다른 정수도 모두 마찬가지다.
    딱 4개의 값은 가능하다.   " 0 , -1 , 0x3f , 0x7f. "

  • size_t 자료형은 '이론상 가장 큰 사이즈를 담을 수 있는 unsigned 데이터 타입'으로 정의된다.

# 2. bzero

void	ft_bzero(void *s, size_t n)
{
   ft_memset(s, 0, n);
}
  • bzero 함수는 어떤 메모리의 시작점(b)에서 연속된 범위(n)만큼 0 으로(바이트 단위) 초기화하는 함수이다.
  • memset 함수를 이용하여 쉽게 만들 수 있다.

# 3. memcpy

void	*ft_memcpy(void *dst, const void *src, size_t n)
{
   size_t			i;
   unsigned char	*cp_dst;
   unsigned char	*cp_src;

   i = 0;
   cp_dst = (unsigned char *)dst;
   cp_src = (unsigned char *)src;
   while (i++ < n)
   	*cp_dst++ = *cp_src++;
   return (dst);
}
  • memcpy 함수는 문자열(src) 의 시작점에서 n 만큼을 dst 에 복사함.
  • src 의 null을 검사하지 않고 n byte 만큼 복사함.
  • memory 영역은 반드시 겹치지 않아야한다. 겹칠 경우, memmove 를 사용해야한다.

    메모리 겹침에 대한 stackoverflow 의견
    https://stackoverflow.com/questions/43088070/meaning-of-overlapping-when-using-memcpy memcpy vs memmove
    https://stackoverflow.com/questions/4415910/memcpy-vs-memmove

  • memcpy() 은 dest 의 포인터 를 반환한다.
  • const void * 의 경우 컴파일 시 오류 문구가 생기므로 형변환 이 필요하다.

# 4. memccpy

void	*ft_memccpy(void *dst, const void *src, int c, size_t n)
{
   size_t			i;
   unsigned char	*cp_dst;
   unsigned char	*cp_src;

   i = 0;
   cp_dst = (unsigned char *)dst;
   cp_src = (unsigned char *)src;
   while (i < n)
   {
   	cp_dst[i] = cp_src[i];
   	if (cp_src[i] == (unsigned char)c)
   		return (dst + i + 1);
   	i++;
   }
   return (0);
}
  • memccpy 함수는 문자열(src) 의 시작점에서 n 만큼을 dst 에 복사하는데 c 값을 찾으면 c 까지만 복사하고 종료한다.
  • return 값이 특이한데, c 값이 n 범위 안 src에 존재한다면, c가 있던 src의 다음 주소 를 반환하며, n 범위 안에 c 값이 없다면 NULL 을 반환한다.
