# —save

2번째 argument로 —save를 넣을 시, bmp 를 만들어야 한다.

# 초기 에러처리

이전에 argc 가 2인 경우에만 가능했는데, 3일 때도 추가해야 한다.

  • argc ==2 일때, argv[1] 은 *.cub 이어야 한다.
  • 또는, argc==3 일때, argv[1] 은 *.cub 이며 argv[2] 는 --save 여야 한다.
  • img.ptr 을 모두 만든 후 save flag 유무에 따라 window를 생성할지 말지 결정하자.

그렇다면 이 조건을 한번에 적을 수 있을까?

# OPEN 함수

  • fcntl.h 헤더에 정의되어 있다.
// 두 함수 호출은 동일하다.
fd = creat(pathname, mode);
fd = open(pathname, O_WRONLY | O_CREAT | O_TRUNC, mode} // mode 는 권한옵션

  • mode : 파일을 생성할때의 권한을 주는 옵션이다.

    만약 0777을 주게 된다면 모두 읽기, 쓰기, 실행 권한을 주게 되는 것이다. 또한 가독성을 높이기 위해 심볼릭 상수도 제공한다. 심볼릭 상수를 사용할때 역시 | 연산으로 권한을 줄 수 있다.

  • 반환값 : 성공적으로 파일을 열게되면 파일 디스크립터(fd)를 반환. 실패하면 -1 반환.

# BMP Header

	ft_bzero(hd, 14);
	hd[0] = 0x42;   // 'B'
	hd[1] = 0x4D;   // 'M'
	size = info->win.w * info->win.h * 4 + 54;
	hd[2] = size % 256;		       // 전체 크기를 역순으로 1byte 단위로 넣자.
	hd[3] = size / 256 % 256;    // 1byte 는 최대 256이므로 256을 나눠주자.
	hd[4] = size / 256 / 256 % 256;
	hd[5] = size / 256 / 256 / 256 % 256;
	hd[10] = 54;	// img 데이터 위치
	write(fd, hd, 14);

# DIB (Device Independent Bitmap) 비트맵 정보 헤더

void	bmp_info_header(t_info *info, int fd)
{
	unsigned char	dib[40];

	ft_bzero(dib, 40);
	dib[0] = (unsigned char)(40);
	dib[4] = (unsigned char)(info->win.w % 256);
	dib[5] = (unsigned char)(info->win.w / 256 % 256);
	dib[6] = (unsigned char)(info->win.w / 256 / 256 % 256);
	dib[7] = (unsigned char)(info->win.w / 256 / 256 / 256);
	dib[8] = (unsigned char)(info->win.h % 256);
	dib[9] = (unsigned char)(info->win.h / 256 % 256);
	dib[10] = (unsigned char)(info->win.h / 256 / 256 % 256);
	dib[11] = (unsigned char)(info->win.h / 256 / 256 / 256);
	dib[12] = (unsigned char)(1);
	dib[14] = (unsigned char)(32);
	write(fd, dib, 40);
}

# BMP DATA

해상도 크기에 따라 padding 값이 생기는데, padding 을 뺀 부분만 출력해주자.

void	bmp_data(t_info *info, int fd)
{
	int				total_len;
	int				line;
	int				padding;
	int				i;
	unsigned char	*buf;

	total_len = info->img.size_line * info->win.h;
	padding = info->img.size_line - info->win.w * 4;
	line = info->img.size_line;
	i = 0;
	while (i < info->win.h)
	{
		buf = (unsigned char *)(info->img.data + total_len - line);
		write(fd, buf, info->img.size_line - padding);
		line += info->img.size_line;
		i++;
	}
}

실제 출력 가능한 크기만 넣어주면 코드가 더 깔끔해진다.

// size_line - padding
write(fd, buf, info->img.size_line - padding);

// or

// 실제 출력 크기
write(fd, buf, info->win.w * 4);

# 7. 학습 내용에 대한 개인적인 총평

  • 비트맵을 c 언어로 만들 수 있다는 것 자체에서 새로운 학습이 되었다. bitmap 을 만들기 위해서 어떠한 값이 필요하며, 그냥 출력시 상하반전된 것처럼 나오기 때문에 값을 역순으로 출력해줘야 하는데 x 방향은 순방향이기 때문에 한줄의 길이를 알아야 한다. size_line 을 미리 받아왔기 때문에 수월한 부분이었지만, mlx 신버전에서는 padding 값이 자동으로 생겨서 padding 부분을 빼준 실제 값 부분만 출력해줘야 정확한 bitmap file을 만들 수 있었다.
  • 이제 sprite 만 끝내면 일단 코드 구현은 끝마친다. 오래도록 해온 cub3d를 이번 달 안에 끝내도록 열심히 학습에 매진하도록 해야겠다.
