# Self Closing 태그 <tag />

리액트에서는 html 태그를 꼭 닫아줘야한다.

<input> 이나 <b> 와 같은 단일 태그는 <input /> <b /> 로 적어서 닫아주도록 하자!

# Fragment <> </>

리액트는 두가지 이상 태그가 있는 경우 꼭 감싸야하는데, <div> 로 감싸거나, Fragment <> </> 를 사용할 수 있다.

function App() {
  return (
    <>
      <Hello />
      <div>안녕히계세요</div>
    </>
  );
}

Fragment 는 브라우저 상에서 따로 별도의 엘리먼트로 나타나지 않는다.

# JSX 안에 변수 사용

{} 로 감싸서 자바스크립트 변수를 사용할 수 있다.

function App() {
	const value = '안녕하세요';
  return (
    <>
      <Hello />
      <div>{value}</div>
    </>
  );
}

# props 사용법

props 는 properties 의 줄임말이다.

컴포넌트에게 값을 전달할 때 사용한다.

App.js

function App() {
  return (
    <Hello name="react" />
  );
}

Hello.js

function Hello(props) {
  return <div>안녕하세요 {props.name}</div>
}

props 는 객체 형태로 전달되며, name 을 조회할 경우 props.name 으로 조회해야 한다.

# 비구조화 할당을 이용해 코드를 간결하게 하자.

함수의 파라미터에서 비구조화 할당 문법을 사용하면 코드가 간결해진다.

// 일반 코드
function Hello(props) {
	return <div style={{color: props.color}}>안녕하세요 {props.user}</div>
}

// 비구조화 할당 문법을 이용한 코드
function Hello({ name, color }) {
  return <div style={{ color }}>안녕하세요 {name}</div>

이때 color 는 color:color 로 적어도 되고, 축약 가능하다.

return <div style=>안녕하세요 {name}</div>

# defaultProps

특정값이 없을 경우, 기본값을 지정해줄 수 있다.

// App.js --------------------
<>
	<Hello user='React' color='blue'/>
	<Hello color='red'/>
</>

// Hello.js -------------------
function Hello({user, color}) {
	return <div style={{color}}>안녕하세요 {user}</div>
}

// Hello.js 의 Props 기본값 설정
Hello.defaultProps = {
	user: '이름없음'
}

function Wrapper({ children }) {

# usaState 를 통한 동적 상태 관리

# Counter.js

숫자 카운트 되는 기본 예제

import React, {useState} from 'react';

function Counter() {
	const [number, setNumber] = useState(0);

	const onIncrease = () => {
		setNumber(number + 1);
	}
	const onDecrease = () => {
		setNumber(number - 1);
	}

	return (
		<div>
			<h1>0</h1>
			{/* 함수를 onIncrease() 로 적지 말자!!- */}
			<button onClick={**onIncrease**}>+1</button>
			<button onClick={**onDecrease**}>-1</button>
		</div>
	);
}

export default Counter;
  • 함수를 onIncrease() 로 적을 경우, 렌더링 될때 함수가 동작한다.
  • 여기선 클릭할 때 실행되어야 하므로 함수타입의 값을 넣어주자.

# useState 사용하기

먼저, react 에서 useState 함수를 불러오자.

import React, {useState} from 'react';

const [number, setNumber] = useState(0);
  • number : 현재상태 (변수)
  • setNumber : 현재 상태를 업데이트하는 (함수)
  • useState(0) : 여기서 0 은 기본 값이다. 5 로 바꾸면 number 의 초기값은 5 가 된다.

# 함수형 업데이트(최적화)

함수형으로 적을 경우가 더 최적화된 형식이다.

const onIncrease = () => {
		setNumber(number + 1);
	}
// 최적화
const onIncrease = () => {
		setNumber(prevNumber => prevNumber + 1);
	}

# INPUT 상태 관리

Input 태그에서 받은 값을 동기적으로 표현하고, 리셋하는 예제

import React, {useState} from 'react';

function InputSample() {
	const [text, setText] = useState('');

	const onChange = (e) => {
		setText(e.target.value);
	}

	const onReset = () => {
		setText('');
	}

	return (
		<div>
			<input onChange={onChange} value={text} />
			<button onClick={onReset}>초기화</button>
			<div>
				<b>값: </b>
				{text}
			</div>
		</div>
	);
}

export default InputSample;

# INPUT 이 여러개일 경우,

각 input 에 해당하는 값을 useState에서 객체로 만들고, <input> 태그의 name 으로 연결지어 준다.

import React, {useState} from 'react';

function InputSample() {
	const [inputs, setInputs] = useState({
		name: '',
		nickname: '',
	});
	const {name, nickname} = inputs;

	const onChange = (e) => {
		const {name, value} = e.target;

		setInputs({
			...inputs, // 원래 객체를 복사해서 값을 바꿔줘야 한다!!
			[name]: value
		});
	};

	const onReset = () => {
		setInputs({
			name: '',
			nickname: '',
		})
	};

	return (
		<div>
			<input
				name="name"
				onChange={onChange}
				placeholder="이름"
				value={name}
			/>
			<input
				name="nickname"
				onChange={onChange}
				placeholder="닉네임"
				value={nickname}
			/>
			<button onClick={onReset}>초기화</button>
			<div>
				<b>값: </b>
				{name} ({nickname})
			</div>
		</div>
	);
}

export default InputSample;

# DOM 선택을 위한 useRef 사용

DOM Selector 함수를 사용하기 위해서 래액트는 ref 를 사용한다.

ref 사용하기 위해선 useRef Hook 함수를 이용해야 한다.

초기화 버튼을 누를시 다시 name <input> 에 포커스가 가도록 만들어 보자.

import React, {useState, useRef } from 'react'; // 1. useRef 불러오기

function InputSample() {
	const [inputs, setInputs] = useState({
		name: '',
		nickname: '',
	});
	const nameInput = useRef(); // 2. useRef 함수 호출
	const {name, nickname} = inputs;

	const onChange = (e) => {
		const {name, value} = e.target;
		setInputs({
			...inputs,
			[name]: value
		});
	};

	const onReset = () => {
		setInputs({
			name: '',
			nickname: '',
		});
		nameInput.current.focus(); // 4. 만든 DOM 객체에 원하는 작업 설정
	};

	return (
		<div>
			<input
				name="name"
				onChange={onChange}
				placeholder="이름"
				value={name}
				ref={nameInput} // 3. 원하는 DOM에 ref 설정하기
			/>
			<input
				name="nickname"
				onChange={onChange}
				placeholder="닉네임"
				value={nickname}
			/>
			<button onClick={onReset}>초기화</button>
			<div>
				<b>값: </b>
				{name} ({nickname})
			</div>
		</div>
	);
}

export default InputSample;
