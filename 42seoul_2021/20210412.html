# 틀린 방법(헤맨 부분)

int 형 배열을 join 할 때, 그냥 join 을 사용하면 아래와 같이 오류가 난다.

import random

a = [random.randint(-100, 100) for r in range(10)]
print(type(a)) # class "list"

result = ' '.join(a)
print(result)
# TypeError: sequence item 0: expected str instance, int found

# 해결 방법

int 값을 str 값으로 변경 후 join 을 해주도록 하자.

import random

num_list = [random.randint(-100, 100) for r in range(10)]
print(" ".join(map(str, num_list)))

하지만 이 방법은 같은 숫자가 나올 수 있어서 다른 방식을 이용하기로 결정함.

# 2번째 방법

# random_sample.py

import random

arr = list(range(-500,500))
#print(arr) -500 ~ 499 의 배열을 만든다
sample_arr = random.sample(arr, 100) # 100개의 중복값 없는 sample 을 추출한다
result = " ".join(map(str, sample_arr)) # ' ' 공백이 있는 문자열로 만든다
print(result)

# random_argv.py

2번째 방법에서 argv 를 추가시킨 파일

# py random_argv.py <시작 범위> <끝 범위(미포함)> <총 개수>
# py random_argv.py -10 10 10
# -10 ~ 9 범위에서 10개를 숫자를 랜덤하게 뽑아 문자열로 생성
import sys
import random

num1 = int(sys.argv[1])
num2 = int(sys.argv[2])
num3 = int(sys.argv[3])

arr = list(range(num1, num2))
# print(arr)
sample_arr = random.sample(arr, num3)
result = " ".join(map(str, sample_arr))
print(result)
