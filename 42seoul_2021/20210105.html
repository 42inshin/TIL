# print_char 처리

  • print str 를 새롭게 다시 malloc 으로 구현
  • tester lover -v2 테스팅(c, s) 완료
int	print_str(va_list ap, t_info *info)
{
	int		cnt;
	char	*value;
	char	*s;
	int		len;
	int 	w;

	cnt = 0;
	value = va_arg(ap, char *);
	// value 가 NULL 인 경우, (null) 로 표시
	if (value == NULL)
		value = "(null)";
	len = ft_strlen(value);
	w = info->width;
	if (len > info->precision && info->precision >= 0)
		len = info->precision;
	if (len > w)
		w = len;
	// 조건에 따라 폭만큼 메모리 할당
	if (!(s = (char *)malloc(sizeof(char) * (w + 1))))
		return (-1);
	s[w] = '\0';
	// 0 인 경우는 생각 안해줘도 되므로 빈 칸(' ')으로 초기화
	ft_memset(s, ' ', w);
	// '-' 여부에 따라 왼쪽 / 오른쪽 정렬
	if (info->minus == 1)
		ft_memcpy(s, value, len);
	else
		ft_memcpy(s + w - len, value, len);
	cnt += ft_putstr(s);
	free(s);
	return (cnt);
}
  • print nbr 를 malloc으로 구현하였으나
    • 값이 음수 일때의 예외처리 아직 덜 진행함.

# 7. 학습 내용에 대한 개인적인 총평

  • malloc 사용 없이 printf 를 구현해보려 했을 때는 조건들이 서로 묶여있어서 if 문을 알맞게 퍼즐처럼 맞춰주는 작업이 필요했다.
  • 하지만, malloc 을 사용하는 코드로 바꾸면서 코드가 더 직관적으로 바뀌게 된 것을 확인할 수 있었다.
  • 다음 nbr 부분도 malloc 으로 구현 중이나, str 과는 다르게 값이 음수인 경우의 예외 처리도 있어서 좀더 시간이 소요될 것으로 보인다.
  • 코딩 테스트를 꾸준히 하려고 하는데 지금 까지는 1단계 레벨을 풀어서 그런지 어렵다는 느낌으 없다. 아마도 지금까지 꾸준히 코딩을 하면서 나름의 성과가 있었던 것 같다. 하루 하루 더 보람차게 코딩을 하면서 사회에 도움이 되는 멋진 작품을 세상에 선보일 수 있도록 노력하자.
