# Truthy, Falsy

  • Falsy 한 값 5 가지 : undefined, null, 0 , '', NaN

    나머지 값들은 모두 true 이다.

console.log(!undefined);
console.log(!0);
console.log(!'');
console.log(!null);
console.log(!NaN);
// 모두 true , 즉 느낌표가 없으면 false 인 값들이다.

# 단축 평가 논리 계산법

true 이면 뒤에껄 출력, false 이면 false 출력한다.

console.log(true && 'hello');       // hello
console.log(false && 'hello');      // false
console.log('hello' && false);      // false
console.log('hello' && 'bye');      // bye
console.log(null && 'hello');       // null
console.log(undefined && 'hello');  // undefined
console.log(0 && 'hello');          // 0
console.log('' && 'hello');         // ''

위 조건을 이용하여 값이 없는 상태의 코드를 줄일 수 있다.

const namelessDog = {
	name : '',
}

function getName(animal) {
	const name = animal && animal.name;
if (name == false) {
		return '이름이 없는 동물입니다.'
	}
	return name;
}

const name = getName(namelessDog);
console.log(name);

색칠한 부분을 아래와 같이 한줄로 작성할 수 있다.

	return name || '이름이 없는 동물입니다.'

// or 연산자의 경우 앞이 true 이면 앞에 값을 리턴하고, 앞의 값이 false 일때는 뒤에 값을 리턴한다.

# 함수의 기본 파라미터 지정

함수에 인자값을 넣어주지 않은 경우, 기본값으로 반환되도록 설정해줄 수 있다.

파라미터에 ( = ) 을 이용하면 기본값 설정을 해줄 수 있다..

function calCircleArea(r = 1) {
	return Math.PI * r * r;
}
// 더 줄이면
const calCircleArea = (r = 1) => Math.PI * r * r;

const area = calCircleArea();
console.log(area);  // 3.141592653589793

# 조건문 더 스마트하게 쓰기

객체를 이용하면 조건문을 보다 더 깔끔하게 쓸 수 있다.

function getSound(animal) {
	if (animal === '개') return '멍멍';
	if (animal === '고양이') return '야옹';
	if (animal === '참새') return '짹짹';
}
console.log(getSound('고양이'));

// 객체로 적을 경우,
function getSound(animal) {
	const sound = {
		개: '멍멍',
		고양이: '야옹~',
		참새: '짹짹'
	}
	return sound[animal] || '...?';
}

console.log(getSound('고양이'));

function makeSound(animal) {
	const tasks = {
		개: function() {
			console.log('멍멍');
		},
		고양이: () => {
			console.log('야옹~');
		},
		참새(){
			console.log('짹짹');
		}
	}
	const task = tasks[animal];
	if(!task) {
		console.log('...?')
		return;
	}
	task();
}
makeSound('고양이');

# 비구조화 할당

const array = [1, 2];

const [one, two = 1] = array; // two = 1 은 기본값을 지정해준 것이다.

console.log(one); // 1
console.log(two); // 2

const object = {a: 1, b: 2};

const [a, b = 1] = object; // b = 1 은 기본값을 지정해준 것이다.

console.log(a); // 1
console.log(b); // 2

# spread

객체, 배열을 복사할 수 있다.

// spread 연산자:  ...
const a = {
	type: '알파벳',
}

const b = {
	...a,            // a 객체의 속성값이 복사된다.
	color: 'blue',
}

console.log(a);  // { type: '알파벳' }
console.log(b);  // { type: '알파벳', color: 'blue' }

# rest

나머지 값을 모아올 수 있다. 이름은 rest 가 아니어도 상관없음.

const a = {
	type: '알파벳',
	color: 'purple',
	attribute: 'cute',
}
const { color, ...rest } = a;

console.log(color); // purple
console.log(rest); // { type: '알파벳', attribute: 'cute' }

# 함수 파라미터에서의 rest

function sum(...rest) {
	return rest.reduce((acc,current) => acc + current, 0);
}

console.log(sum(1,2,3,4,5,6));  // 21

# 문제 ) max 값 구하기

// 주어진 값 중에서 가장 큰값을 반환하라.
function max(...rest) {
	let tmp;
	for (let i = 0; i < rest.length - 1; i++) {
		if (rest[i] > rest[i + 1]) {
			tmp = rest[i]
		}
	}
	return tmp;
}

const result = max(1, 2, 3, 4, 10, 5, 6, 7);
console.log(result);

답

function max(...rest) {
	return	rest.reduce((pre, current) => pre > current ? pre : current)
}

const result = max(1, 2, 3, 4, 10, 5, 6, 7);
console.log(result);
