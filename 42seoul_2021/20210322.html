# 함수 선언 방법

  • 기본 방식
function sum(a, b) {
	return a + b;
}
console.log(sum(1,2));
  • ES6 문법 : 화살표 함수
const sum = (a, b) => {
	return a + b;
}
console.log(sum(1, 2));

# 객체 비구조화 할당

  • 기본 문법
const ironMan = {
	name: '토니 스타크',
	actor: '로버트 다우니 주니어',
	alias: '아이언맨'
}

function print(hero)
{
	// 일반 문법
	const text = `${hero.alias}(${hero.name}) 역할을 맡은 배우는 ${hero.actor}이다.`
	console.log(text);
}

print(ironMan)
  • 객체 - 비구조화 할당
function print(hero)
{
	// 비구조화 할당
	const { alias, name, actor } = hero;
	// 템플릿 리터럴(Template Literal) 문법 ` `
	const text = `${alias}(${name}) 역할을 맡은 배우는 ${actor}이다.`
	console.log(text);
}

# Getter 와 Setter 함수

  • Getter : 특정 값을 조회할때마다 호출하는 함수
const numbers = {
	a: 1,
	b: 2,
	get sum() {
		console.log('sum 함수가 실행됩니다!');
		return this.a + this.b;
	}
};

console.log(numgers.sum);
// 결과
// sum 함수가 실행됩니다.
// 3
  • Setter : 특정 값으로 변경(설정)할 때 호출하는 함수
const dog = {
	_name: '멍멍이',
	get name() {
		return this._name;
	}
	set name(value) {
		console.log('이름이 바뀝니다..' + value);
		this._name = value;
	}
}
console.log(dog.name); // 멍멍이 ._name 으로 검색해야되나,
dog.name = '뭉뭉이';     // getter 함수를 같은 이름으로 만들어서 .name 으로 ._name 값을 받아옴
console.log(dog.name); // 뭉뭉이

getter 함수와 setter 함수는 같은 이름으로 넣어줄 수 있다.

  • 예시
const numbers = {
	_a: 1,
	_b: 2,
	sum: 3,
	calculate() {
		document.write('calculate');
		this.sum = this._a + this._b;
	},
	get a() {
		return this._a;
	},
	get b() {
		return this._b;
	},
	set a(value) {
		this._a = value;
		this.calculate();
	},
	set b(value) {
		this._b = value;
		this.calculate();
	}
};

getter, setter 함수를 사용했을 때 위의 예시의 장점은 값을 바꿀 때만 계산하게 해줘서 추가로 호출할 때에 불필요한 연산을 하지 않는 것이다.

# 객체 keys, values

객체의 key와 value 값을 Object 객체의 method (프로퍼티에 할당된 함수)로 받아올 수 있다.

const doggy = {
	name: '멍멍이',
	sound: '멍멍',
	age: 2
};

console.log(Object.entries(doggy)); // 둘 다
console.log(Object.keys(doggy));    // key
console.log(Object.values(doggy));  // value

# 배열 내장 함수

# forEach

배열 안, 각각의 값을 호출할때 사용

const cats = ["나쵸", "라라"];
// 기본 함수
cats.forEach(function(e) {
  console.log(e);
});
// 화살표 함수
cats.forEach((e) => {
  console.log(e);
});
// 결과
// 나쵸
// 라라

# map

배열 안의 각각의 값을 변환할 때 사용

const arr = [1, 2, 3];

const square = n => n * n; // 화살표 함수
const result = arr.map(square);
console.log(result);

//또는
const result2 = arr.map(n => n * n);
console.log(result2);
// 결과
// [1, 4, 9]

# indexOf

배열 안 값을 알고 있을 때, 인덱스 구하는 함수

값을 못 찾은 경우, -1 반환

const arr = ['하나', '둘', '셋', '넷'];

const result = arr.indexOf('셋');
console.log(result);
// 결과 : 2

# findIndex

배열 안에 조건식에 부합하는 값을 찾아 인덱스를 반환하는 함수

조건에 맞는 값 중 첫번째 값을 리턴

const arr = [
	{	id: 1, content: '안녕' },
	{	id: 2, content: '하이' },
	{	id: 3, content: '헬로' }
];

const result = arr.findIndex(a => a.id === 3);
console.log(result);
// 결과 : 2

# find

배열 안에 조건식에 부합하는 값을 찾아 객체를 반환하는 함수

const objects = arr.find(a => a.id === 3);
console.log(objects);
// 결과
// {	id: 3, content: '헬로' }

# filter

조건에 맞는 모든 값을 배열에 담아 리턴

const todos = [
	{	id:1, done:true },
	{	id:2, done:true },
	{	id:2, done:false	}
];
const todoDone = todos.filter(todo => todo.done == true);
console.log(todoDone);

# splice

index 에서 n 개만큼 자르고, 자른 값을 리턴하는 함수

const num = [10, 20, 30, 40];
const index = num.indexOf(30);
const result = num.splice(index, 2);

console.log(result);
console.log(num);
// 결과
// [30, 40] : result
// [10, 20] : index

# slice

index 에서 n 개만큼을 리턴한다. (원래 값을 자르지는 않음!)

const num = [10, 20, 30, 40];
const index = num.indexOf(30);
const result = num.slice(index, 2);

console.log(result);
console.log(num);
// 결과
// [30, 40] : result
// [10, 20, 30, 40] : index

# shift, pop, unshift, push

  • shift : 앞에서부터 값을 빼준다.
  • unshift : 앞에서부터 값을 넣어준다.
  • pop : 뒤에서부터 값을 빼준다.
  • push : 뒤에서부터 값을 넣어준다.

push() 예시

const array = [1, 2, 3];
array.push(100);
console.log(array);
// 결과
[1, 2, 3, 100]

pop() 예시

const array = [1,2,3];
array.shift();
console.log(array);
// 결과
[2, 3];

# concat

배열을 합쳐주는 함수

기존의 배열을 수정하지 않는다.

const arr1 = [1,2,3];
const arr2 = [4,5,6];

const concated = arr1.concat(arr2);
console.log(concated);
//결과
// [1,2,3,4,5,6];

# join

배열 사이 사이에 값을 넣어준다.

const arr = [1,2,3,4,5,6];

const joined = arr.join('번');
console.log(joined);
// 결과
// 1번2번3번4번5번6

# reduce

배열의 값들을 더해줄 수 있다.

const arr = [1, 2, 3, 4, 5];
//합값: 합해지는 값, 현재값:arr의 각 값들, 0 : 합값의 초기값
const sum = arr.reduce((합값, 현재값) => 합값 + 현재값, 0);
console.log(sum);
// 결과: 15

아래는 forEach 로 같은 동작을 하는 함수를 구현한 것이다.

let sum = 0;
arr.forEach((e) => {
	sum += e;
})
console.log(sum);
// 결과: 15

# reduce 로 평균을 구하기

const arr = [1, 2, 3, 4, 5];

const avg = arr.reduce((sum, current, index, array) => {
	if (index === array.length - 1)
		return (sum + current) / array.length;
	return sum + current;
}, 0);  // sum 의 초기값이다.
console.log(avg);
// 결과: 3
  • sum: 합친 값
  • current: 현재 값
  • index: 각 값의 index
  • array:자기 자신
  • 0: sum의 default value

# reduce 로 공통 값 세기

const alpa = ['a','a','a','b','c','c','d','e'];

const count = alpa.reduce((acc, current) => {
	if (acc[current]) {
		acc[current] += 1;
	} else {
		acc[current] = 1;
	}
	return acc;
}, {});
console.log(count);
// 결과: {a: 3, b: 1, c: 2, d: 1, e: 1}a: 3b: 1c: 2d: 1e: 1__proto__: Object
  • acc: 더해지는 변수, 초기값은 {} 빈 객체로 설정
  • current: alpa의 각각의 값
  • acc[current] == acc['a'] == acc.a 이므로 빈 객체에 없으면 추가되고 있으면 1씩 증가
