# 클래스

객체 생성자와 동일한 기능을 쉽게 만들 수 있는 문법이다.

class Animal {
	constructor(type, name, sound) {
		this.type = type;
		this.name = name;
		this.sound = sound;
	}
	say() {
		console.log(this.sound);
	}
}
const dog = new Animal('개', '바둑이', '멍멍');

dog.say();

constructor: 생성자, 객체의 초기화를 담당한다.

# extends 상속

상속 받는 기능도 extents 와 super 로 쉽게 만들 수 있다.

class Dog extends Animal {
	constructor(name, sound) {
		super('개', name, sound);
	}
}
const dog = new Dog('개', '바둑이', '멍멍');
dog.say();
  • 클래스 활용 예시
class Nature {
	constructor(name) {
		this.name = name;
		this.objects = [];
	}
	addObject(value) {
		this.objects.push(value);
	}
	print() {
		const arr = this.objects.join(', ');
		console.log(`${this.name} 중에는 ${arr}가 있다.`);
	}
}

const a = new Nature('포유류');
a.addObject('인간');
a.addObject('강아지');
a.addObject('고양이');
a.addObject('원숭이');

console.log(a.print());
// 포유류 중에는 인간, 강아지, 고양이, 원숭이가 있다.

# static

클래스 함수 자체에 메서드를 설정할 수 있다.

class A {
	static name = 'B';
	static age = 37;
	static hello() {
		console.log(A.age);
	}
}

console.log(A, A.age); // [class B] 37

# override

부모에 있는 변수를 자식에도 같은 이름으로 만들어 주면 그것을 오버라이딩 이라고 한다. 자식이 부모를 덮어씌운다.

class Parent {
	name = 'Lee';
	hello() {
		console.log(this.name);
	}
}

class Child extends Parent {
	age = 12;
	hello() {
		console.log(this.name, this.age);
	}
}

const p = new Parent();
const c = new Child();
console.log(p); // Parent { name: 'Lee' }
console.log(c); // Child { name: 'Lee', age: 12 }

# super

자식이 constructor를 할 때 부모 요소도 이용하고 싶은 경우 super를 꼭 추가해줘야 한다.

class Parent {
	name;
	constructor(name) {
		this.name = name;
	}
	hello() {
		console.log(this.name);
	}
}

class Child extends Parent {
	age;
	constructor(name, age) {
		super(name);
		this.age = age;
	}
	hello() {
		console.log(this.name, this.age);
	}
}

const p = new Parent('Mark');
const c = new Child('Mark', 37);
console.log(p, c);
// Parent { name: 'Mark' }
// Child { name: 'Mark', age: 37 }

# Promise

생성자를 통해서 프로미스 객체를 만들 수 있다.

생성자의 인자로 executor 라는 함수를 이용한다.

executor 함수는 resolve 와 reject 를 인자로 가진다.

(resolve, reject) ⇒ { ... };

생성자를 통해서 프로미스 객체를 만드는 순간 pending (대기) 상태에 돌입하게 된다.

new Promise((resolve, reject) => {}); // Pending

executor 함수 인자 중 하나인 resolve 함수를 실행하면, fulfilled (이행) 상태가 된다.

new Promise((resolve, reject) => {
	// 함수 동작
	resolve(); // fulfilled
});

executor 함수 인자 중 하나인 reject 함수를 실행하면, rejected (거부) 상태가 된다.

new Promise((resolve, reject) => {
	// 함수 동작
	reject(); // rejected
});

예시, p 라는 프로미스 객체는 1000ms 후에 fulfilled 됩니다.

const p = new Promise((resolve, reject) => {
	// pending
	setTimeout(() => {
		resolve(); // 1초 후 fulfilled
	}, 1000);
});

p.then(() => { /* callback */ });

fulfilled 되면서 callback 이 실행됩니다.

p.then(() => {
	console.log('1000ms 후에 fulfilled 됩니다.');
});

then 을 설정하는 시점을 명확하게 하고, 함수의 실행과 동시에 프로미스 객체를 만들면서 pending 이 시작하도록 하기 위해 프로미스 객체를 생성하면서 리턴하는 함수 (p) 를 만들어 함수(p) 실행과 동시에 then 을 설정합니다.

function p() {
	return new Promise((resolve, reject) => {
		// pending
		setTimeout(() => {
			resolve(); // 1초 후 fulfilled
		}, 1000);
	});
}
p().then(()=>{
	console.log('1000ms 후에 fulfilled 됩니다.')
});

reject 는 catch() 를 통해 실행된다.

function p() {
	return new Promise((resolve, reject) => {
		// pending
		setTimeout(() => {
			reject(); // rejected
		}, 1000);
	});
}
p()
	.then(()=>{
	console.log('1000ms 후에 fulfilled 됩니다.')
	})
	.catch(()=>{
	console.log('1000ms 후에 rejected 됩니다.')
	});

excutor 의 resolve 함수를 실행할때 인자를 넣어 실행하면, then 의 callback 함수의 인자로 받을 수 있다.

function p() {
	return new Promise((resolve, reject) => {
		// pending
		setTimeout(() => {
			resolve('hello');
		}, 1000);
	});
}
p().then((message) => {
	console.log('1000ms 후에 fulfilled 됩니다.', message)
});
// 1000ms 후에 fulfilled 됩니다. hello

reject 함수의 경우, catch 의 callback 함수의 인자로 받을 수 있다.

p().then((message) => {
	console.log('1000ms 후에 fulfilled 됩니다.', message)
}).catch((reason) => {
	console.log('1000ms 후에 rejected 됩니다.', reason)
});

reject의 경우 보통 Error 객체를 만들어서 보내 준다.

function p() {
	return new Promise((resolve, reject) => {
		// pending
		setTimeout(() => {
			reject(new Error('bad'));
		}, 1000);
	});
}
p().then((message) => {
	console.log('1000ms 후에 fulfilled 됩니다.', message);
}).catch((error) => {
	console.log('1000ms 후에 rejected 됩니다.', error);
});
/* 결과
1000ms 후에 rejected 됩니다. Error: bad
    at Timeout._onTimeout (/Users/ninos/Desktop/js_sample/script.js:6:11)
    at listOnTimeout (internal/timers.js:554:17)
    at processTimers (internal/timers.js:497:7)
 */

fulfilled 되거나 rejected 된 후에 최종적으로 실행할 것이 있다면, .finally() 를 설정하고, 함수를 인자로 넣는다.

p().then((message) => {
	console.log('1000ms 후에 fulfilled 됩니다.', message);
}).catch((error) => {
	console.log('1000ms 후에 rejected 됩니다.', error);
}).finally(() => {
	console.log('끝났다.')
});

# Callback

보통 비동기 작업을 할때, callback 함수를 인자로 넣어 로직이 끝나면 callback 함수를 호출합니다. 이런 경우 함수가 아래로 진행되지 않고, callback 함수 안으로 진행됩니다.

// callback 함수 기본 구조
function c(callback) {
	setTimeout(() => {
		callback();
	}, 1000);
}

c(() => {
	console.log('1000ms 후에 callback 함수가 실행됩니다.')
});

c(() => {
	c(() => {
		c(() => {
			console.log('3000ms 후에 callback 함수가 실행됩니다.')
		});
	});
});

# Promise 체이닝

callback 함수처럼 여러번 불러올 경우 괄호 안에 괄호를 적으면서 callback 지옥을 경험 할 수 있는데, prmise 의 경우, .then() 을 통하여 연결해주면서 여러번 불러올 수 잇다.

function p() {
	return new Promise((resolve, reject) =>{
		setTimeout(() => {
			resolve();
		}, 1000);
	});
}
// promise 체이닝
p()  // 아래 3개는 다 같은 방법이다.
.then(()=>{
	return p();
})
.then(() => p())
.then(p)
.then(() => {
	console.log('4초 후 실행됩니다.')
});

value 가 프로미스 객체인지 아닌지 알 수 없는 경우,

  • value 가 프로미스 객체이면, resolve 된 then 메서드를 실행함.
  • value 가 프로미스 객체가 아니면, value 를 인자로 보내면서 then 메서드를 실행함.
Promise.resolve(new Promise((resolve, reject) => {
	setTimeout(() => {
		resolve('foo');
	}, 1000);
})).then((data) => {
	console.log(`프로미스 객체인 경우, resolve 된 결과를 받아 then 이 실행 됩니다.`, data);
})

Promise.resolve('bar').then(data => {
	console.log('then 메서드가 없는 경우, fulfilled 됩니다.', data);
})

// 결과 : 위아래 바껴서 나옴.
// then 메서드가 없는 경우, fulfilled 됩니다. bar
// 프로미스 객체인 경우, resolve 된 결과를 받아 then 이 실행 됩니다. foo

# Promise.all

여러개의 프로미스를 병렬로 처리하고 싶은 경우에 주로 사용한다.

프로미스 객체 여러개를 생성하여, 배열로 만들어 인자로 넣고 Promise.all 을 실행하면, 배열의 모든 프로미스 객체들이 fulfilled 되었을 때, then 의 함수가 실행된다.

then 의 함수의 인자로 프로미스 객체들의 resolve 인자값을 [배열]로 돌려줍니다.

# Promise.race

여러개의 프로미스 객체 중 가장 먼저 끝난(fulfilled) 하나가 반환됩니다.

# async await

비동기처리가 순차적으로 흘러가는 것이 장점이다.

await는 async function 안에서만 사용할 수 있다.

function p(ms) {
	return new Promise((resolve, reject) =>{
		setTimeout(() => {
			resolve();
		}, ms);
	});
}

async function main() {
	const ms = await p(1000);
	console.log(`${ms} 가 실행 된다.`);
}

# try, catch

async 함수 내에서 resolve, reject를 나누려면 try, catch() 를 이용하면 된다.

async function main() {
	try {
		const ms = await p(1000);
		console.log(`${ms} 가 실행 된다.`);
	} catch(error) {
		console.log(error);
	} finall
}

# finally

finally 를 사용하면, 메인 어싱크 함수가 모두 실행 되고 나서 다른 동작을 실행 시킬 수 있다.

async function main() {
	try {
		const ms = await p(1000);
		console.log(`${ms} 가 실행 된다.`);
	} catch(error) {
		console.log(error);
	} finally {
		console.log('the end')
	}
}

# await Promise.all / race

async, await 에서도 promise all 을 사용할 수 있다. 전체가 다 완료되면 결과를 반환한다.(동시에 = 병렬로 실행 된다.)

async function main() {
	const results = await Promise.all([p(1000), p(2000), p(3000)]);
	console.log(results);
}

  • 전날 밤부터 편두통이 심해져서 전날 밤 시간과 오늘 오전 시간까지 모두 날린 상태가 되었다. 자고 나서 조금은 나아졌으나 여전히 몸 상태가 좋지 않아, 학습에 대한 활력이 떨어지는 상태이다. 정신적으로 압박감이 많아지는 요즘이다.

  • c 언어를 놓은지 그렇게 오래되지 않은 것 같은데, 벌써부터 기억이 가물가물해지는게 느껴져서 큰일이다. 둘다 놓치지 않도록 계속해서 신경쓰고 얼른 이것을 마치고 cub3d 도 다시 시작하도록 해야겠다.
