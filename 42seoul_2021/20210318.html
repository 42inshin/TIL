# 백엔드 설정

백엔드는 개발환경과 배포환경을 동일하게 해줘야 차후에 서버를 변경 시 오류를 방지할 수 있다.

그러므로 python 의 venv 를 이용하여 가상 환경을 만들어서 서버를 구축하도록 하자.

개발환경과 배포환경을 동일하게 하라! 백엔드는 쉽게 오류가 날 수 있다.

# 서버는 가상환경이 필수다!!!

pip list 를 쓰면 설치된 패키지와 버전을 알 수 있다.

각각의 패키지 버전을 다맞춰줘야한다.

그러므로 파이썬의 venv 를 이용하여 가상환경을 만들어주자.

# 가상환경 생성
python -m venv <가상환경 이름>

# 폴더 생성된다.

# 가상환경 실행
source <가상환경 이름>/bin/activate

# django 설치

  1. pip install django 를 통해 장고를 설치하자.
  2. pip list 로 설치 확인

# 프로젝트 생성

# django-admin startproject <프로젝트 이름> <경로>
django-admin startproject MBIT .
# 기본 데이터베이스 테이블 적용
python manage.py migrate

MBIT 폴더가 생성 되면 [settings.py](http://settings.py) 문서에서 ALLOWED_HOSTS 값을 추가하자.

# 모든 사용자 허용
ALLOWED_HOSTS = ['*']

# 서버 실행

# IP 0 의 80포트에 서버를 실행한다.
python manage.py runserver 0:80

서버가 잘 작동되는지 확인을 하려면 프로젝트→실행 URL과 포트 에서 링크를 열어준다.

확인후 Ctrl + c 로 서버 종료.

# 모델 만들기

모델이란 부가적인 메타데이터를 가진 데이터베이스의 구조(layout)를 말한다.

자세한 사항은 공식문서 참조 (opens new window)

models.py 작성

from django.db import models

# Create your models here.
class Developer(models.Model):
    name = models.CharField(max_length=50)
    count = models.IntegerField(default=0)

class Question(models.Model):
    number = models.IntegerField(unique=True)
    content = models.CharField(max_length=100)

class Choice(models.Model):
    content = models.CharField(max_length=100)
    question = models.ForeignKey(to='main.Question',
                                 on_delete=models.CASCADE)
    developer = models.ForeignKey(to='main.Developer',
                                  on_delete=models.CASCADE,
                                 null=True)

# Migration 생성

모델을 추가했기 때문에 기존의 DB 데이터에서 새롭게 migration 해줘야 한다.

keyword: ORM(Object Relational Mapping, 객체-관계 매핑)

mysql, mariaDB 등의 데이터베이스 설치 없이 Django 만으로 데이터베이스를 만드는 기법.

  • migration 생성
python manage.py makemigrations
  • DB에 생성한 migration 반영
python manage.py migrate

from django.contrib import admin
from .models import Developer, Question, Choice
# admin에 등록
admin.site.register(Developer)
admin.site.register(Question)
admin.site.register(Choice)

서버를 끄지않고 파일 변경해도 등록이 된것을 확인 할 수 있다.

# 데이터 이름 설정하기

위에서 만든 데이터베이스에서 값을 입력하면 Object (1) 이런 식으로 이름이 생성된다.

이름을 바꾸기 위해선 models.py에서 아래처럼 추가 코드를 입력해줘야 한다.

from django.db import models

# Create your models here.
class Developer(models.Model):
    name = models.CharField(max_length=50)
    count = models.IntegerField(default=0)

    def __str__(self):
        return self.name

class Question(models.Model):
    number = models.IntegerField(unique=True)
    content = models.CharField(max_length=100)

    def __str__(self):
        return f'{self.number}. {self.content}'

class Choice(models.Model):
    content = models.CharField(max_length=100)
    question = models.ForeignKey(to='main.Question',
                                 on_delete=models.CASCADE)
    developer = models.ForeignKey(to='main.Developer',
                                  on_delete=models.CASCADE,
                                 null=True)
    def __str__(self):
        return self.content

# views.py 설정

index.html 을 넣기 위해서 먼저 views.py에 코드를 추가해야한다.

데이터를 받아서 데이터 값을 화면에 뿌려주는 동적인 페이지를 만들기 위해서 설정하는 것이다.

from django.shortcuts import render
from .models import Question, Developer, Choice

def index(request):
    developers = Developer.objects.all()

    context = {
		    # 'developers'는 html 사용하는 변수이고, 뒤의 developers 는 데이터 값이다.
		    'developers' : developers,
    }

    return render(request, 'index.html', context=context)

# CSRF 공격을 방어하자

CSRF 공격(Cross Site Request Forgery)은 웹 어플리케이션 취약점 중 하나로 인터넷 사용자(희생자)가 자신의 의지와는 무관하게 공격자가 의도한 행위(수정, 삭제, 등록 등)를 특정 웹사이트에 요청하게 만드는 공격이다.

출처 : https://itstory.tk/entry/CSRF-공격이란-그리고-CSRF-방어-방법 (opens new window) [덕's IT Story]

django 에서는 csrf_token 으로 CSRF 를 막을 수 있다.

HTML

태그 영역에 아래의 코드를 추가하자.

<!-- 이 코드가 없을 경우, 아예 작동을 안한다 -->
<form action="/result/" method="post">
	{% csrf_token %}
	...

# 변수 넣어주기

<form action="/result/" method="post">
      {% csrf_token %}
      {% for question in questions %}
	      <h3>{{ question.number }}번 {{ question.content }}</h3>
	      {% for choice in question.choice_set.all %}
	          <div>
	              <input type="radio" name="question-{{ question.pk }}" id="choice-{{ choice.pk }}" value="{{ choice.developer.pk }}">
	              <!-- <label for="choice-{{ choice.pk }}">{{ forloop.counter }}. {{ choice.content }}</label> -->
	          </div>
		     {% endfor %}
          <hr>
      {% endfor %}
      <input type="submit" value="제출하기">
</form>

.pk 는 migrations 안에 있는 0001_initial.py 에서 ID에서 고유한 값을 가져온다.

{{ forloop.counter }}

는 템플릿 태그에 있는 값이다. for 가 지나갈때마다 1씩 증가.
